\begin{frame}  
  \frametitle{LF12cd::Kontext:Wasserfall}

   \textbf{Verfahren:} 
   
    \vspace{0.2cm}
    \begin{itemize}
      \item \textbf{Projektmanagerin} $\rightarrow$ \textit{Funktionalen Design}: \textbf{Scopebeschreibung}
      \item \textbf{Funktionales Design} $\rightarrow$ \textit{Technisches Design}: \textbf{funktionale Vorgaben}
      \item \textbf{Technisches Design} $\rightarrow$ \textit{Programmierung}: \textbf{technische Vorgaben}
      \item \textbf{Programmierung} $\rightarrow$ \textit{Testen}: \textbf{Testrelease}
      \item \textbf{Testen} $\rightarrow$ \textit{Deployment}: \textbf{Deploymentrelease}
    \end{itemize}

    \vspace{0.3cm}
    \begin{footnotesize}
    \textbf{Probleme:} 
    
    \vspace{0.1cm}
    \begin{itemize}
      \item lange abgeschlossene Zyklen 
      \item spät erkennbarer Änderungsbedarf
      \item formalistisch langwieriges Wiederansetzen
      \item aufwendiges Restaffing
      \item große 'Reibungsverluste' beim Verstehen
    \item oft von Wirklichkeit / Markt überholt
    \end{itemize}

    \end{footnotesize}
\end{frame}
