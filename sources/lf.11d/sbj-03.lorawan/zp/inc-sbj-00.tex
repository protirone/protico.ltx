\begin{frame}  
  \frametitle{LF11d:LoRaWan}


  LoRaWan 

  \begin{itemize}
    \item steht für Long Range Wide Area Network
    \item ist ein Low-Power-Wireless-Netzprotokoll auf der Ebene der Vermittlungsschicht 
    \item hat frei verfügbare Spezifikationen und Software-Grundmodule als Open-Source-Software 
    \item nutzt aber das proprietäre, patentierte, leitungslose „LoRa“-Übertragungsverfahren auf der Bitübertragungsschicht 
  \end{itemize}

  \vspace{1cm}
  \begin{tiny}
  \begin{flushright}
    vgl. \href{https://de.wikipedia.org/wiki/Long\_Range\_Wide\_Area_Network}{https://de.wikipedia.org/wiki/Long\_Range\_Wide\_Area\_Network}
  \end{flushright}
  \end{tiny}

\end{frame}


\begin{frame}  
  \frametitle{LF11d:LoRaWan:Besonderheiten}

  Lora ist/hat/bietet

  \begin{itemize}
    \item ein leitungsloses Übertragungsverfahren auf der Bitübertragungsschicht,
    \item asymmetrisch,
    \item auf Energieeffizienz für Reichweiten über 10 km ausgerichtet
    \item eine Datenübertragungsrate reicht von 292 bit/s bis 50 kbit/s.
  \end{itemize}
\end{frame}
