\begin{frame}  
  \frametitle{LF11d.HTTP:Methods/Commands}

  \begin{itemize}
    \item \textcolor{teal}{GET} = zum Anfordern einer Ressource
    \item \textcolor{teal}{POST} = zur Übermittlung von zu verarbeitenden Daten
    \item \textcolor{teal}{HEAD} = zur Anfrage des Antwortheaders ohne Body 
    \item \textcolor{teal}{PUT} = zur Übermittlung von serverseitig zu speichernden Daten
    \item \textcolor{teal}{PATCH} = zur Änderung von serverseitig existierenden Dateien
    \item \textcolor{teal}{DELETE} = zum Anfordern einer serverseitigen Löschung 
    \item \textcolor{teal}{TRACE} = zum Abfragen, wie die eigene Anfrage serverseitig ankommt
    \item \textcolor{teal}{OPTIONS} = zum Abfrage serverseitig unterstützter Methoden
    \item \textcolor{teal}{CONNECT} = (für Proxyserver)
  \end{itemize}
\end{frame}
