\begin{frame}  
  \frametitle{LF11C:02::AP2.PRC:FMEA:Und jetzt?}

  \underline{F}ailure \underline{M}ode and \underline{E}ffects \underline{A}nalysis:

  \vspace{0.2cm}  
  \begin{itemize}
    \item bilde interdisziplinäres Team
    \item beschreibe System-/Prozessstruktur
    \item liste für jede Komponente mögliche Fehler auf
    \item bewerte jeden Fehler (je [1 - 10]) hinsichtlich seiner
    \begin{itemize}
      \item \textbf{B}edeutung (\textbf{S}everity)
      \item \textbf{A}uftretenswahrscheinlichkeit (\textbf{O}ccurence)
      \item \textbf{E}ntdeckungswahrscheinlichkeit (\textbf{D}etection)
    \end{itemize}
    \item bilde für jeden Fehler die Risiko-Prioritätszahl \textbf{RPZ} mittels \texttt{B*A*E}
    \item kümmere Dich als Nächstes um den Fehler mit der höchsten RPZ
  \end{itemize}

\end{frame}

\begin{frame}  
  \frametitle{LF11C:02::AP2.PRC:ABC-Analyse:Und jetzt?}

 \begin{footnotesize}
  \begin{itemize}
    \item Ermittle alle Fehler
    \item  Ermittle für jeden Fehler:
      \begin{itemize}  
      \item den absoluten Schadenswert
      \item seine Häufigkeit
      \item den relativen Schadenswert (Häufigkeit*abs.Schadenswert)
    \end{itemize}
    \item Addiere alle relativen Schadenswerte zum Gesamtschaden
    \item Ermittle für jeden relativen Schadenswert den Anteil am Gesamtschaden
    \item Erzeuge eine Liste von Wertpaaren 'Fehler:Anteil am Gesamtschaden'
    \item Sortiere die Liste absteigend
    \item Gruppiere sie in drei Cluster (Klassen):
    \begin{itemize}
      \item \textbf{Klasse A}: die ersten Fehler, die zusammen 80\% des Schadens bilden.
      \item \textbf{Klasse B}: die nächsten Fehler, die zusammen 15\% des Schadens bilden.
      \item \textbf{Klasse C}: die letzten Fehler, die zusammen 5\% des Schadens bilden
    \end{itemize}
    \item Kümmere Dich als zuerst um die Klasse A Fehler
  \end{itemize}
\end{footnotesize}
\end{frame}

\begin{frame}  
  \frametitle{LF11C:02::AP2.PRC:Nutzwertanalyse:Und jetzt?}

 \begin{tiny}
  \begin{itemize}
    \item Liste Alternativen auf
    \item Liste Bewertungskriterien auf
    \item Gewichte Kriterien nach Wichtigkeit im Vergleich zu anderen Kriterien
    \item Ermittle für jeden relativen Schadenswert den Anteil am Gesamtschaden
    \item Ermittle für jede Alternative für jedes Kriterium den entsprechenden Wert
    \item Multipliziere für jede Alternative den Kriteriumswert mit dem Gewicht
    \item Multipliziere Kriteriumswerte einer Alternative zum Gesamtnutzen der Alternative
  \end{itemize}
  \end{tiny}
