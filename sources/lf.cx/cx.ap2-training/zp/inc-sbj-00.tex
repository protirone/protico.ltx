\begin{frame}  
  \frametitle{LF.CX:AP2-Strategie:Prozess}

  \vspace{0.5cm}
  \begin{enumerate}
    \item Jeden \textcolor{violet}{Aufgabensatz* knallig} einfärben.
    \item \textcolor{cyan}{Direkte Ergänzungen} dazu \textcolor{cyan}{positiv-dezent} einfärben.
    \item \textcolor{teal}{Hintergrundinformationen} dazu am Rand \textcolor{teal}{positiv-sanft} anstreichen.
    \item Quick-Wins einsammeln.
    \item Dicke Bretter bohren.
  \end{enumerate}

  
  \vspace{0.5cm}
  \begin{tiny}
  \begin{flushright}
    Aufgabensätze beginnen mit \textit{[Ergänzen | Erläutern | Beschreiben | Berechnen | Benennen | Ordnen | ... ] Sie ... }
  \end{flushright}
  \end{tiny}

\end{frame}
