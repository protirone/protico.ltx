\begin{frame}  
  \frametitle{LF.CX:AP2-Strategie:Prozess}

  \vspace{0.5cm}
  \begin{enumerate}
    \item \textbf{\textcolor{violet}{Grobscan}}: Jeden \textcolor{violet}{Aufgabensatz* knallig} einfärben.
    \item Feinklärung zu jedem Aufgabensatz:
    \begin{itemize}
      \item {Direkte Ergänzungen} dazu \textcolor{cyan}{positiv-dezent} einfärben.
      \item Entferntere \textcolor{teal}{Hintergrundinformationen} dazu \textcolor{teal}{positiv-sanft} markieren.
    \end{itemize}
    
    \item Quick-Wins einsammeln.
    \item Dicke Bretter bohren.
  \end{enumerate}

  
  \vspace{0.5cm}
  \begin{tiny}
  \begin{flushright}
    *) Aufgabensätze beginnen mit \textit{[Ergänzen | Erläutern | Beschreiben | Berechnen | Benennen | Ordnen | ... ] Sie ... }
  \end{flushright}
  \end{tiny}
